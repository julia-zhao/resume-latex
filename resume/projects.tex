\makebox[\projlen][l]{\entrytitlestyle{Image Concatenator}}\
\cvtag{C++}\
\cvtag{Concurrency}\
\cvtag{Asynchronous I/O}\
\cvtag{Semaphores}\
\begin{adjustwidth*}{0.3cm}{0cm}
\descriptionstyle{Multithreaded program consisting of producer and consumer threads. Producers can request for PNG image slices from a server, and store the result in a shared fixed-size buffer after validating the contents. Consumer threads may then copy buffer data to shared memory to concatenate each slice into a full image.}
\end{adjustwidth*}
